```rust
async fn test_cache_effectiveness() {
    // 第一次处理（缓存未命中）
    let result1 = pipeline.process_document(document).await.unwrap();

    // 第二次处理（缓存命中）
    let result2 = pipeline.process_document(document).await.unwrap();

    assert!(result2.cache_hits > 0, "第二次应该有缓存命中");

}
```
