```rust
pub trait LLMProviderV3: Send + Sync {
    fn name(&self) -> &str;

    async fn chat_stream(&self, req: ChatRequest) -> Result<Stream<String>, LLMError>;
    fn count_tokens(&self, text: &str) -> usize;
    async fn validate(&self) -> Result<bool, LLMError>;
}
```

### 2. 工具自动重试
```rust
// 自动重试机制（最多 3 次）
let mut attempts = 0;
let max_attempts = 3;

loop {
    attempts += 1;
    match tool.execute(params.clone()).await {
        Ok(output) => return Ok(output),
        Err(e) if attempts < max_attempts => {
            tokio::time::sleep(Duration::from_millis(100 * attempts as u64)).await;
        }
        Err(e) => return Err(e),
    }
}
```

### 3. 智能模型切换

## 🚀 使用示例

### 多模型切换
```rust
use newclaw::*;

#[tokio::main]
async fn main() -> Result<()> {
    // 创建 OpenAI Provider
    let openai = OpenAIProvider::new("your-api-key".to_string());
    
    // 创建 Claude Provider
    let claude = ClaudeProvider::new("your-api-key".to_string());
    
    // 定义策略：成本优化
    let strategy = ModelStrategy::CostOptimized {
        cheap: "gpt-4o-mini".to_string(),
        premium: "gpt-4o".to_string(),
    };
    
    let model = strategy.select(100);
    
    let request = ChatRequest {
        messages: vec![Message {
            role: MessageRole::User,
            content: "Hello!".to_string(),
            tool_calls: None,
            tool_call_id: None,
        }],
        model: model.clone(),
        temperature: 0.7,
        max_tokens: Some(1000),
        top_p: None,
        stop: None,
        tools: None,
    };
    
    let response = openai.chat(request).await?;
    println!("{}", response.message.content);
    
    Ok(())
}
```

```rust
async fn main() -> Result<()> {

```

### 流式响应
```rust
use newclaw::*;
use newclaw::llm::streaming::*;

#[tokio::main]
async fn main() -> Result<()> {
    let provider = OpenAIProvider::new("your-api-key".to_string());
    
    let request = ChatRequest {
        messages: vec![/* ... */],
        model: "gpt-4o-mini".to_string(),
        temperature: 0.7,
        max_tokens: Some(1000),
        top_p: None,
        stop: None,
        tools: None,
    };
    
    stream_llm_response(&provider, request, |chunk| {
        match chunk {
            StreamChunk::Data(data) => {
                println!("Chunk: {}", data);
            }
            StreamChunk::Done => {
                println!("Stream complete");
            }
            StreamChunk::Error(e) => {
                eprintln!("Error: {}", e);
            }
        }
    }).await?;
    
    Ok(())
}
```
