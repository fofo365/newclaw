## 🎯 核心模块

### 1. 事件轮询系统 (`polling.rs`)
```rust
use newclaw::feishu_websocket::{EventPoller, PollingConfig};

// 创建轮询器
let config = PollingConfig::default();
let mut poller = EventPoller::new(config, ws_config, handler);

// 启动
poller.start().await?;

// 检查状态
let queue_size = poller.queue_size().await;
let is_running = poller.is_running().await;

// 停止
poller.stop().await?;
```

### 2. 消息类型支持 (`messages.rs`)
```rust
use newclaw::feishu_websocket::{
    TextMessage, RichTextMessage, CardMessage,
    ImageMessage, FileMessage, MessageSender
};

let sender = MessageSender::new("https://open.feishu.cn");

// 文本消息
let text = TextMessage::new("Hello!")
    .with_root_id("parent_msg_id");  // 回复消息

// 富文本消息
let rich = RichTextMessage::new(
    RichTextContent::new()
        .with_title("标题")
        .add_paragraph(vec![
            RichTextParagraph::Text {
                text: "内容".to_string(),
                style: Some(vec![TextStyle::Bold]),
            },
        ])
);

// 卡片消息
let card = CardMessage::new(
    CardContent::new()
        .with_header("卡片标题")
        .with_header_template("blue")
        .add_element(CardElement::Div {
            text: Some(CardText::plain("内容")),
            fields: None,
            extra: None,
        })
);

// 发送
sender.send_text("chat_id", ReceiveIdType::ChatId, text).await?;
```

### 3. 错误重试机制 (`retry.rs`)
```rust
use newclaw::feishu_websocket::{
    RetryExecutor, RetryStrategy, RetryManager,
    CacheFallback, DefaultValueFallback, AlertRule
};

// 策略配置
let strategy = RetryStrategy {
    max_attempts: 5,
    initial_delay: Duration::from_secs(1),
    max_delay: Duration::from_secs(60),
    multiplier: 2.0,
    jitter: true,
    ..Default::default()
};

// 基本重试
let executor = RetryExecutor::new(strategy);
let result = executor.execute(|| async {
    // 可能失败的操作
    Ok("success")
}).await?;

// 带降级的重试
let fallback = Arc::new(DefaultValueFallback::new("默认值"));
let executor = RetryExecutor::new(strategy).with_fallback(fallback);

// 带监控的重试
let manager = RetryManager::new(strategy)
    .add_alert_rule(AlertRule::new("告警", ErrorSeverity::High, 5, 60))
    .with_alert_callback(|rule, metrics| {
        println!("告警触发: {}", rule.name);
    });

let result = manager.execute_with_metrics(|| async {
    // 操作
}).await?;

// 查看指标
let metrics = manager.get_metrics().await;
```

## 📋 常用配置

### PollingConfig
```rust
PollingConfig {
    polling_interval: Duration::from_secs(5),      // 轮询间隔
    long_polling_timeout: Duration::from_secs(30), // 长轮询超时
    max_queue_size: 1000,                          // 队列大小
    max_concurrent_handlers: 10,                   // 并发数
    event_processing_timeout: Duration::from_secs(60), // 处理超时
    enable_long_polling: true,                     // 启用长轮询
    max_retries: 3,                                // 重试次数
    retry_delay: Duration::from_secs(1),          // 重试延迟
}
```

### RetryStrategy
```rust
RetryStrategy {
    max_attempts: 3,                              // 最大尝试
    initial_delay: Duration::from_secs(1),       // 初始延迟
    max_delay: Duration::from_secs(60),          // 最大延迟
    multiplier: 2.0,                              // 退避倍数
    jitter: true,                                 // 启用抖动
    jitter_range: 0.1,                            // 抖动范围
}
```
