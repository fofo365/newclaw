```rust
pub struct ToolRegistry {
    tools: HashMap<String, Box<dyn Tool>>,
}

impl ToolRegistry {
    pub fn register(&mut self, tool: Box<dyn Tool>) {
        let name = tool.name().to_string();
        self.tools.insert(name, tool);
    }
    
    pub async fn execute(&self, name: &str, params: Value) -> Result<ToolOutput> {
        match self.tools.get(name) {
            Some(tool) => tool.execute(params).await,
            None => Err(Error::ToolNotFound(name.to_string())),
        }
    }
}
```

```rust
pub enum ModelStrategy {
    Static(String),
    RoundRobin(Vec<String>),
    Fallback { primary: String, fallback: String },
    CostOptimized { cheap: String, premium: String },
}

impl ModelStrategy {
    pub async fn select(&self) -> String {
        match self {
            ModelStrategy::Static(model) => model.clone(),
            ModelStrategy::RoundRobin(models) => self.next_model(models),
            ModelStrategy::Fallback { primary, fallback } => {
                if self.is_healthy(primary).await { primary.clone() } else { fallback.clone() }
            },
            ModelStrategy::CostOptimized { cheap, premium } => {
                if self.is_complex_task() { premium.clone() } else { cheap.clone() }
            },
        }
    }
}
```

### 3. 流式响应 🟡 P1

**目标**: 实时流式输出，提升体验

#### 3.1 SSE 支持
```rust
// src/gateway/streaming.rs
use axum::response::Sse;

pub async fn chat_stream_handler(
    State(agent): State<Arc<AgentEngine>>,
    Json(req): Json<ChatRequest>,
) -> Sse<impl Stream<Item = Result<Event, Infallible>>> {
    let stream = agent.chat_stream(req).await.unwrap();
    
    Sse::new(stream.map(|chunk| {
        Ok(Event::default().data(chunk))
    }))
    .keep_alive(
        axum::response::sse::KeepAlive::new()
            .interval(Duration::from_secs(1))
            .text("keepalive"),
    )
}
```

#### 3.2 WebSocket 流式
```rust
pub async fn handle_websocket_stream(
    ws: WebSocketUpgrade,
    State(agent): State<Arc<AgentEngine>>,
) -> Response {
    ws.on_upgrade(|socket| async move {
        let stream = agent.chat_stream(req).await.unwrap();
        
        while let Some(chunk) = stream.next().await {
            socket.send(Message::Text(chunk)).await.unwrap();
        }
    })
}
```

#### 3.3 Feishu 流式支持
```rust
// Feishu 不支持流式，需要分块发送
pub async fn send_streaming_chunks(&self, chunks: Vec<String>) {
    for (i, chunk) in chunks.iter().enumerate() {
        if i == 0 {
            self.send_message(chunk).await;
        } else {
            self.edit_message(message_id, chunk).await;
        }
        tokio::time::sleep(Duration::from_millis(100)).await;
    }
}
```

### 4. Agent 协作引擎 🟡 P1

**目标**: 多 Agent 协作（基于 v0.2.0 的通信基础设施）

#### 4.1 Agent 发现
```rust
pub struct AgentRegistry {
    agents: HashMap<String, AgentInfo>,
}

impl AgentRegistry {
    pub async fn discover(&mut self, addr: SocketAddr) {
        // 通过 WebSocket 发现其他 Agent
        let client = WebSocketClient::connect(format!("ws://{}", addr)).await?;
        let info = client.handshake().await?;
        self.agents.insert(info.id.clone(), info);
    }
}
```

#### 4.2 任务分发
```rust
pub struct TaskDispatcher {
    agents: Arc<RwLock<HashMap<String, AgentInfo>>>,
}

impl TaskDispatcher {
    pub async fn dispatch(&self, task: Task) -> Result<TaskResult> {
        // 根据能力选择 Agent
        let agent = self.select_agent_by_capability(&task.capability).await?;
        
        // 发送任务
        let msg = InterAgentMessage {
            payload: MessagePayload::Request(Request::ExecuteTask(task)),
            ..Default::default()
        };
        
        agent.send(msg).await?;
        
        // 等待结果
        let result = agent.recv().await?;
        Ok(result)
    }
}
```
